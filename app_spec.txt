<project_specification>
  <project_name>open-autocoder</project_name>

  <overview>
    A TypeScript/React port of the autonomous coding agent system, integrated with opencode instead of Claude Code.
    Enables long-running AI agents to build complete applications across multiple sessions with parallel execution,
    feature tracking, dependency-aware scheduling, and real-time monitoring. Developers can run it via npm/npx,
    as a system service, or as a desktop application.
  </overview>

  <error_handling_philosophy>
    <principle name="never_swallow">All unexpected errors must be surfaced to the user, never silently ignored</principle>
    <principle name="actionable_messages">Error messages must explain what went wrong AND what the user can do about it</principle>
    <principle name="severity_levels">Distinguish blocking/fatal errors from warnings and informational messages</principle>
    <principle name="always_log">Even handled errors must be logged for debugging purposes</principle>
    <principle name="fail_fast">Do not continue execution in an invalid state; fail early and visibly</principle>
    <principle name="structured_errors">Use typed error classes with error codes, messages, and suggested actions</principle>
    <anti_patterns>
      <bad>catch (e) { /* ignore */ }</bad>
      <bad>catch (e) { console.log("error") }</bad>
      <bad>.catch(() => {})</bad>
      <bad>|| true to suppress failures</bad>
    </anti_patterns>
    <good_patterns>
      <example>
        catch (e) {
          logger.error("Failed to connect to opencode", { error: e });
          throw new UserFacingError(
            "Cannot connect to opencode server",
            "Make sure opencode is running: `opencode serve`"
          );
        }
      </example>
    </good_patterns>
  </error_handling_philosophy>

  <database_schema>
    <tables>
      <features>
        - id: INTEGER PRIMARY KEY AUTOINCREMENT
        - priority: INTEGER NOT NULL (lower = higher priority)
        - category: TEXT NOT NULL (e.g., "functional", "style", "security")
        - name: TEXT NOT NULL
        - description: TEXT NOT NULL
        - steps: TEXT NOT NULL (JSON array of step strings)
        - passes: INTEGER NOT NULL DEFAULT 0 (boolean: 0=false, 1=true)
        - in_progress: INTEGER NOT NULL DEFAULT 0 (boolean)
        - dependencies: TEXT (JSON array of feature IDs, nullable)
        - created_at: TEXT NOT NULL (ISO 8601 timestamp)
        - updated_at: TEXT NOT NULL (ISO 8601 timestamp)
      </features>
      <projects>
        - name: TEXT PRIMARY KEY
        - path: TEXT NOT NULL (absolute path to project directory)
        - created_at: TEXT NOT NULL
        - updated_at: TEXT NOT NULL
        - model: TEXT (preferred AI model)
        - provider: TEXT (preferred AI provider)
        - concurrency: INTEGER DEFAULT 3
        - yolo_mode: INTEGER DEFAULT 0 (boolean)
        - testing_agent_ratio: INTEGER DEFAULT 1
      </projects>
      <providers>
        - id: TEXT PRIMARY KEY (e.g., "openai", "anthropic")
        - name: TEXT NOT NULL (display name)
        - api_key_encrypted: TEXT (encrypted API key)
        - base_url: TEXT (custom API endpoint)
        - is_default: INTEGER DEFAULT 0 (boolean)
        - created_at: TEXT NOT NULL
        - updated_at: TEXT NOT NULL
      </providers>
      <sessions>
        - id: TEXT PRIMARY KEY (UUID)
        - project_name: TEXT NOT NULL (FK to projects)
        - agent_type: TEXT NOT NULL (initializer, coding, testing)
        - feature_id: INTEGER (FK to features, nullable)
        - started_at: TEXT NOT NULL
        - ended_at: TEXT (nullable)
        - status: TEXT NOT NULL (running, completed, failed, cancelled)
        - output_log: TEXT (path to log file)
      </sessions>
    </tables>
  </database_schema>
